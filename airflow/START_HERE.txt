# Step 1: Navigate to airflow directory
cd c:\Users\hp\OneDrive\Desktop\DataCamp\covid_project\airflow

# Step 2: Install Python dependencies
pip install -r requirements.txt

# Step 3: Initialize Airflow database
airflow db upgrade

# Step 4: Create admin user (if first time)
airflow users create --username admin --password admin --firstname Admin --lastname Admin --role Admin --email admin@example.com

# Step 5: Configure environment
copy .env.example .env
# Edit .env with your AWS and Snowflake credentials

# Step 6: Run setup script
.\setup_airflow_env.ps1 `
  -AwsAccessKey "your_aws_key" `
  -AwsSecretKey "your_aws_secret" `
  -SnowflakeUser "your_sf_user" `
  -SnowflakePassword "your_sf_password" `
  -SnowflakeAccount "your_sf_account"

##############################################################################
#  ▶️  START AIRFLOW (2 Terminals)
##############################################################################

# Terminal 1 - Scheduler (runs DAGs on schedule)
airflow scheduler

# Terminal 2 - Webserver (web UI for monitoring)
airflow webserver --port 8080
